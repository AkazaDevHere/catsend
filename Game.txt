    <script>
        function sendToDiscord() {
            const name = document.getElementById("name").value;
            const thing1 = document.getElementById("thing1").value;
            const thing2 = document.getElementById("thing2").value;
            const thing3 = document.getElementById("thing3").value;
            const thing4 = document.getElementById("thing4").value;
            const thing5 = document.getElementById("thing5").value;

            // Replace 'YOUR_DISCORD_WEBHOOK_URL' with the actual webhook URL provided by Discord
            const webhookURL = 'YOUR_DISCORD_WEBHOOK_URL';

            // Format the data to be sent to Discord
            const data = {
                content: `Name: ${name}\nThing 1: ${thing1}\nThing 2: ${thing2}\nThing 3: ${thing3}\nThing 4: ${thing4}\nThing 5: ${thing5}`,
            };

            // Send the data to Discord via a webhook
            fetch(webhookURL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(data),
            })
            .then(response => {
                if (response.ok) {
                    alert('Data sent to Discord successfully!');
                } else {
                    alert('Failed to send data to Discord. Please check the webhook configuration.');
                }
            })
            .catch(error => {
                console.error('Error:', error);
            });
        }
    </script>
